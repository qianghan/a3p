function nextBusinessDay(d: Date): Date {
  const r = new Date(d);
  const day = r.getDay(); // 0=Sun, 6=Sat
  if (day === 6) r.setDate(r.getDate() + 2);
  else if (day === 0) r.setDate(r.getDate() + 1);
  return r;
}

app.post('/api/v1/agentbook-core/cashflow/scenarios', async (req, res) => {
  try {
    const tenantId = (req as any).tenantId;
    const { name, description, scenarioType, params, startMonth } = req.body;

    // Run simulation immediately to cache lastResult
    let lastResult: any = null;
    try {
      const simRes = await fetch(`http://localhost:${process.env.PORT || 4050}/api/v1/agentbook-core/simulate`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'x-tenant-id': tenantId },
        body: JSON.stringify({ scenarioType, ...params }),
      });
      if (simRes.ok) lastResult = (await simRes.json()).data;
    } catch { /* best-effort */ }

    const scenario = await db.abCashflowScenario.create({
      data: {
        tenantId, name, description: description || null, scenarioType,
        params: params || {}, startMonth: startMonth ? new Date(startMonth) : null,
        lastResult,
      },
    });
    res.status(201).json({ success: true, data: scenario });
  } catch (err) { res.status(500).json({ success: false, error: String(err) }); }
});

function isBearerValid(req: NextRequest): boolean {
  const secret = process.env.CRON_SECRET;
  if (!secret) return true;
  const auth = req.headers.get('authorization');
  if (!auth) return false;
  const a = Buffer.from(auth), b = Buffer.from(`Bearer ${secret}`);
  return a.length === b.length && timingSafeEqual(a, b);
}

async function computeSnapshotForTenant(tenantId: string) {
  const since30 = new Date(Date.now() - 30 * 24 * 60 * 60 * 1000);
  const config = await db.abTenantConfig.findUnique({ where: { tenantId } });

  // Cash accounts (type='asset', code starts with '1')
  const cashAccounts = await db.abAccount.findMany({
    where: { tenantId, accountType: 'asset', code: { startsWith: '1' }, isActive: true },
    select: { id: true },
  });
  const cashAccountIds = cashAccounts.map(a => a.id);

  // Current cash balance = sum of credits - debits on cash accounts
  const cashLines = await db.abJournalLine.aggregate({
    where: { tenantId, accountId: { in: cashAccountIds } },
    _sum: { creditCents: true, debitCents: true },
  });
  const cashBalance = (cashLines._sum.creditCents ?? 0) - (cashLines._sum.debitCents ?? 0);

  // 30d inflow = credits on cash accounts in last 30 days
  const inflowLines = await db.abJournalLine.aggregate({
    where: { tenantId, accountId: { in: cashAccountIds }, createdAt: { gte: since30 } },
    _sum: { creditCents: true },
  });
  const inflow30d = inflowLines._sum.creditCents ?? 0;

  // 30d outflow = debits on cash accounts in last 30 days
  const outflowLines = await db.abJournalLine.aggregate({
    where: { tenantId, accountId: { in: cashAccountIds }, createdAt: { gte: since30 } },
    _sum: { debitCents: true },
  });
  const outflow30d = outflowLines._sum.debitCents ?? 0;

  // Monthly burn = outflow30d
  const monthlyBurn = outflow30d;
  const runwayMonths = monthlyBurn > 0 ? cashBalance / monthlyBurn : 999;

  // Simple projections: extrapolate monthly net burn
  const netMonthly = inflow30d - outflow30d;
  const proj30 = cashBalance + netMonthly;
  const proj60 = cashBalance + 2 * netMonthly;
  const proj90 = cashBalance + 3 * netMonthly;

  // Tax reserve estimate
  const taxEst = await db.abTaxEstimate.findFirst({
    where: { tenantId }, orderBy: { createdAt: 'desc' },
  });
  const taxReserveBalance = taxEst?.estimatedTaxCents ?? 0;

  await db.abFinancialSnapshot.upsert({
    where: { tenantId_snapshotDate: { tenantId, snapshotDate: new Date(new Date().toDateString()) } },
    update: {
      cashBalanceCents: cashBalance, monthlyBurnCents: monthlyBurn, runwayMonths,
      inflow30dCents: inflow30d, outflow30dCents: outflow30d,
      projectedCash30Cents: proj30, projectedCash60Cents: proj60, projectedCash90Cents: proj90,
      taxReserveBalanceCents: taxReserveBalance,
    },
    create: {
      tenantId, snapshotDate: new Date(new Date().toDateString()),
      cashBalanceCents: cashBalance, monthlyBurnCents: monthlyBurn, runwayMonths,
      arBalanceCents: 0, apBalanceCents: 0, netIncomeMtdCents: 0,
      inflow30dCents: inflow30d, outflow30dCents: outflow30d,
      projectedCash30Cents: proj30, projectedCash60Cents: proj60, projectedCash90Cents: proj90,
      taxReserveBalanceCents: taxReserveBalance,
    },
  });

  return { cashBalance, inflow30d, outflow30d, runwayMonths };
}

export async function GET(request: NextRequest): Promise<NextResponse> {
  if (!isBearerValid(request)) {
    return NextResponse.json({ success: false, error: 'Unauthorized' }, { status: 401 });
  }
  const tenants = await db.abExpense.groupBy({ by: ['tenantId'] });
  const results: Record<string, any> = {};
  for (const { tenantId } of tenants) {
    try {
      results[tenantId] = await computeSnapshotForTenant(tenantId);
    } catch (err) {
      reportError(`[cashflow-snapshot] tenant ${tenantId}`, err);
      results[tenantId] = { error: String(err) };
    }
  }
  return NextResponse.json({ success: true, data: results });
}

    import React, { useEffect, useState } from 'react';
import { LineChart, Line, XAxis, YAxis, CartesianGrid, Tooltip, ResponsiveContainer } from 'recharts';

const SCENARIO_TYPES = [
  { value: 'add_expense', label: 'Add monthly expense' },
  { value: 'add_revenue', label: 'Add monthly revenue' },
  { value: 'lose_client', label: 'Lose a client' },
  { value: 'hire', label: 'Hire someone' },
  { value: 'buy_equipment', label: 'Buy equipment' },
];

export const ScenarioPlanner: React.FC = () => {
  const [scenarios, setScenarios] = useState<any[]>([]);
  const [base, setBase] = useState<any>(null);
  const [newForm, setNewForm] = useState({ name: '', scenarioType: 'hire', params: { monthlyCostCents: 500000 } });
  const [creating, setCreating] = useState(false);

  const coreBase = window.__NAAP_CONFIG__?.coreApiBase || 'http://localhost:4050/api/v1/agentbook-core';
  const token = window.__NAAP_AUTH_TOKEN__ || '';
  const H = { 'Content-Type': 'application/json', Authorization: `Bearer ${token}` };

  useEffect(() => {
    Promise.all([
      fetch(`${coreBase}/cashflow/scenarios`, { headers: H }).then(r => r.json()),
      fetch(`${coreBase}/cashflow/projection`, { headers: H }).then(r => r.json()),
    ]).then(([sc, pr]) => {
      if (sc.success) setScenarios(sc.data);
      if (pr.success) setBase(pr.data.base);
    });
  }, []);

  const handleCreate = async () => {
    setCreating(true);
    const res = await fetch(`${coreBase}/cashflow/scenarios`, {
      method: 'POST', headers: H, body: JSON.stringify(newForm),
    });
    const data = await res.json();
    if (data.success) setScenarios(prev => [data.data, ...prev]);
    setCreating(false);
  };

  const handleDelete = async (id: string) => {
    await fetch(`${coreBase}/cashflow/scenarios/${id}`, { method: 'DELETE', headers: H });
    setScenarios(prev => prev.filter(s => s.id !== id));
  };

  const comparisonData = base ? [
    { label: 'Now', base: 0, scenario: 0 },
    { label: '+30d', base: base.d30, scenario: 0 },
    { label: '+60d', base: base.d60, scenario: 0 },
    { label: '+90d', base: base.d90, scenario: 0 },
  ] : [];

  return (
    <div className="p-6 space-y-6">
      <h1 className="text-2xl font-bold">Scenario Planner</h1>

      <div className="bg-card border border-border rounded-lg p-4">
        <h2 className="text-sm font-semibold mb-3">New Scenario</h2>
        <div className="grid grid-cols-2 gap-3">
          <input placeholder="Scenario name" value={newForm.name}
            onChange={e => setNewForm(f => ({ ...f, name: e.target.value }))}
            className="border border-border rounded px-3 py-2 text-sm bg-background" />
          <select value={newForm.scenarioType}
            onChange={e => setNewForm(f => ({ ...f, scenarioType: e.target.value }))}
            className="border border-border rounded px-3 py-2 text-sm bg-background">
            {SCENARIO_TYPES.map(t => <option key={t.value} value={t.value}>{t.label}</option>)}
          </select>
          <input placeholder="Monthly cost/revenue (cents, e.g. 500000 = $5000)"
            type="number" value={newForm.params.monthlyCostCents}
            onChange={e => setNewForm(f => ({ ...f, params: { monthlyCostCents: Number(e.target.value) } }))}
            className="border border-border rounded px-3 py-2 text-sm bg-background" />
          <button onClick={handleCreate} disabled={creating || !newForm.name}
            className="bg-primary text-primary-foreground rounded px-4 py-2 text-sm font-medium disabled:opacity-50">
            {creating ? 'Running...' : 'Run Scenario'}
          </button>
        </div>
      </div>

      <div className="space-y-3">
        {scenarios.length === 0 && <p className="text-muted-foreground text-sm">No scenarios yet. Create one above.</p>}
        {scenarios.map(s => (
          <div key={s.id} className="bg-card border border-border rounded-lg p-4">
            <div className="flex items-start justify-between">
              <div>
                <p className="font-semibold text-sm">{s.name}</p>
                <p className="text-xs text-muted-foreground">{SCENARIO_TYPES.find(t => t.value === s.scenarioType)?.label}</p>
                {s.lastResult && <p className="text-xs text-muted-foreground mt-1">Impact: 90d balance ${((s.lastResult?.projectedBalance90d ?? 0) / 100).toFixed(0)}</p>}
              </div>
              <button onClick={() => handleDelete(s.id)} className="text-xs text-muted-foreground hover:text-destructive">Remove</button>
            </div>
          </div>
        ))}
      </div>
    </div>
  );
};
